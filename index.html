Template::Plugin::SVG::TT::Graph Demonstration

Template::Plugin::SVG::TT::Graph::Pie

[% cols = [ 'Jan', 'Feb', 'Mar' ] -%]
[% data = [ 12, 45, 21 ] -%]
[% USE graph = SVG.TT.Graph.Pie ( {
    'height' => '600',
    'width'  => '500',
    'show_graph_title'        => 1,
    'graph_title'             => 'Sales in Q1/2002'
    'show_key_data_labels'    => 1,
    'show_key_actual_values'  => 0,
    'show_key_percent'        => 1,
    'rollover_values'         => 1,
    'key'                     => 1,
    'key_placement'           => 'B',
    'expand_greatest'         => 1,
    'fields' => cols }
) -%]

[% graph.add_data({ 'data'  => data, 'title' => 'Sales 2002' }) -%]
[% graph.burn %]

Template::Plugin::SVG::TT::Graph::bar

[% data02 = [ 12, 45, 21 ] -%]
[% data03 = [ 33, 14, 38 ] -%]
[% USE graph = SVG.TT.Graph.Bar ( {
    'height' => '600' ,
    'width'  => '500',
    'show_graph_title'        => 1,
    'graph_title'             => 'Sales in Q1 2002/2003',
    'fields'                  => cols,
    'show_x_labels'           => 1,
    'stagger_x_labels'        => 1,
    'rotate_x_labels'         => 1,
    'show_y_labels'           => 1,
    'show_x_title'            => 1,
    'x_title'                 => 'Months',
    'show_y_title'            => 1,
    'y_title'                 => 'Sales in thousands',
    'show_graph_title'        => 1,
    'key'                     => 1,
    'key_placement'           => 'B'
    }) -%]
[% graph.add_data({ 'data'  => data02, 'title' => 'Sales 2002' }) -%]
[% graph.add_data({ 'data'  => data03, 'title' => 'Sales 2003' }) -%]
[% graph.burn %]
